#include <stdio.h>
#include <stdint.h>

unsigned int invertBits32(unsigned int value) {
    const unsigned int mask = 0xFFFFFFFF; 
    return (~value) & mask;
}

int main() {
    int testCases;
    scanf("%d", &testCases);
    
    for (int i = 0; i < testCases; i++) {
        unsigned int number;
        scanf("%u", &number);
        printf("%u\n", invertBits32(number));
    }
    
    return 0;
}

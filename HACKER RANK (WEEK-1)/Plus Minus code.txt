#include <stdio.h>

void plusMinus(int arr[], int n) {
    int posCount = 0, negCount = 0, zeroCount = 0;

    for (int i = 0; i < n; i++) {
        if (arr[i] > 0)
            posCount++;
        else if (arr[i] < 0)
            negCount++;
        else
            zeroCount++;
    }
    printf("%.6f\n", (float)posCount / n);
    printf("%.6f\n", (float)negCount / n);
    printf("%.6f\n", (float)zeroCount / n);
}

int main() {
    int n;
    scanf("%d", &n);  

    int arr[n];       

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(arr, n);

    return 0;
}
